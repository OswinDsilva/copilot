/**
 * SQL Override Functions
 * Pure functions that generate SQL for specific query patterns
 */

import { getCurrentYear, buildDateFilter, buildShiftFilter } from './dateUtils';
import { PATTERNS, extractMonthNumber } from './patterns';

export function ordinalRowSQL(params: Record<string, any>, query: string): string | null {
  if (!params.row_number) return null;
  
  const rawTable = query.match(PATTERNS.TABLE_NAME)?.[1]?.trim() || '';
  const normalized = rawTable ? rawTable.toLowerCase().replace(/[^a-z0-9]+/g, '_') : '';
  const allowed = new Set(['production_summary', 'trip_summary_by_date', 'uploaded_files', 'equipment']);
  
  if (!normalized || !allowed.has(normalized)) return null;
  
  const offset = Math.max(0, params.row_number - 1);
  return `SELECT * FROM ${normalized} ORDER BY date ASC LIMIT 1 OFFSET ${offset}`;
}

export function equipmentCombinationSQL(params: Record<string, any>): string {
  const limit = params.n || 10;
  let dateFilter = '';
  
  if (params.month) {
    const year = getCurrentYear();
    dateFilter = `WHERE EXTRACT(MONTH FROM trip_date) = ${params.month} AND EXTRACT(YEAR FROM trip_date) = ${year}`;
  } else if (params.date) {
    dateFilter = `WHERE trip_date = '${params.date}'`;
  }
  
  return `SELECT tipper_id, excavator, SUM(trip_count) AS total_trips FROM trip_summary_by_date ${dateFilter} GROUP BY tipper_id, excavator ORDER BY total_trips DESC LIMIT ${limit}`;
}

export function topBottomShiftsSQL(params: Record<string, any>, isBottom: boolean): string {
  const orderDir = isBottom ? 'ASC' : 'DESC';
  const limit = params.n || 10;
  const dateFilter = buildDateFilter(params);
  
  return `SELECT date, shift, qty_ton, total_trips FROM production_summary ${dateFilter} ORDER BY qty_ton ${orderDir} LIMIT ${limit}`;
}

export function shiftSpecificSQL(params: Record<string, any>): string {
  const dateFilter = buildDateFilter(params);
  const shiftFilter = buildShiftFilter(params, !!dateFilter);
  
  return `SELECT date, shift, qty_ton, total_trips, excavator, dumper FROM production_summary ${dateFilter} ${shiftFilter} ORDER BY date DESC`;
}

export function dateRangeAggregationSQL(params: Record<string, any>): string {
  const dateFilter = buildDateFilter(params);
  const shiftFilter = buildShiftFilter(params, !!dateFilter);
  
  return `SELECT date, shift, SUM(qty_ton) AS total_tonnage, SUM(total_trips) AS total_trips FROM production_summary ${dateFilter}${shiftFilter} GROUP BY date, shift ORDER BY date DESC`;
}

export function shiftComparisonSQL(params: Record<string, any>): string {
  const dateFilter = buildDateFilter(params);
  const shiftFilter = buildShiftFilter(params, !!dateFilter);
  
  return `SELECT shift, SUM(qty_ton) AS total_tonnage, SUM(total_trips) AS total_trips, AVG(qty_ton) AS avg_tonnage FROM production_summary ${dateFilter}${shiftFilter} GROUP BY shift ORDER BY total_tonnage DESC`;
}

export function aggregationSummarySQL(params: Record<string, any>, query: string, hasEquipmentUtilization: boolean): string | null {
  const monthNum = params.month || extractMonthNumber(query);
  if (!monthNum) return null;
  
  const year = getCurrentYear();
  let sql = `SELECT 
    SUM(qty_ton) AS total_tonnage,
    SUM(total_trips) AS total_trips,
    AVG(qty_ton) AS avg_daily_tonnage,
    COUNT(DISTINCT date) AS production_days`;
  
  if (hasEquipmentUtilization) {
    sql += `,
    AVG(excavator) AS avg_excavators,
    AVG(dumper) AS avg_dumpers`;
  }
  
  sql += `
  FROM production_summary
  WHERE EXTRACT(MONTH FROM date) = ${monthNum} AND EXTRACT(YEAR FROM date) = ${year}`;
  
  return sql;
}

export function monthlySummarySQL(params: Record<string, any>, query: string): string | null {
  const monthNum = params.month || extractMonthNumber(query);
  if (!monthNum) return null;
  
  const year = getCurrentYear();
  return `SELECT 
    SUM(qty_ton) AS total_tonnage,
    SUM(total_trips) AS total_trips,
    AVG(qty_ton) AS avg_daily_tonnage,
    MAX(qty_ton) AS max_daily_tonnage,
    MIN(qty_ton) AS min_daily_tonnage,
    COUNT(DISTINCT date) AS production_days,
    AVG(excavator) AS avg_excavators,
    AVG(dumper) AS avg_dumpers
  FROM production_summary
  WHERE EXTRACT(MONTH FROM date) = ${monthNum} AND EXTRACT(YEAR FROM date) = ${year}`;
}

export function equipmentSpecificProductionSQL(params: Record<string, any>, query: string): string | null {
  const equipmentIds = params.equipment_ids;
  if (!equipmentIds || equipmentIds.length === 0) return null;
  
  // Determine what metric is being requested
  const isTrips = /trips?/i.test(query);
  const isTonnage = /tonnage|tons?|production/i.test(query);
  
  // Build WHERE clause for equipment IDs
  const equipmentFilter = equipmentIds.length === 1
    ? `WHERE excavator = '${equipmentIds[0]}'`
    : `WHERE excavator IN (${equipmentIds.map((id: string) => `'${id}'`).join(', ')})`;
  
  // Add date filtering if present
  const dateFilter = buildDateFilter(params);
  const combinedFilter = dateFilter 
    ? `${equipmentFilter} AND ${dateFilter.replace('WHERE ', '')}`
    : equipmentFilter;
  
  // Add shift filtering if present
  const shiftFilter = buildShiftFilter(params, true).trim();
  const finalFilter = shiftFilter 
    ? `${combinedFilter} ${shiftFilter}`
    : combinedFilter;
  
  // If asking for trips specifically, sum trip_count
  if (isTrips && !isTonnage) {
    return `SELECT 
      excavator,
      SUM(trip_count) AS total_trips,
      COUNT(DISTINCT trip_date) AS active_days
    FROM trip_summary_by_date
    ${finalFilter}
    GROUP BY excavator
    ORDER BY total_trips DESC`;
  }
  
  // For tonnage/production, we need to estimate or join
  // Since trip_summary_by_date doesn't have tonnage, we'll calculate estimated tonnage
  return `SELECT 
    excavator,
    SUM(trip_count) AS total_trips,
    ROUND(SUM(trip_count * 15.0), 2) AS estimated_tonnage,
    COUNT(DISTINCT trip_date) AS active_days,
    ROUND(AVG(trip_count), 2) AS avg_trips_per_day
  FROM trip_summary_by_date
  ${finalFilter}
  GROUP BY excavator
  ORDER BY total_trips DESC`;
}
