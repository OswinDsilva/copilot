// Pre-compile regex cache for performance (avoids creating RegExp on every check)
const KEYWORD_REGEXES = new Map<string, RegExp>();
 
// Build regex cache at module load time
for (const intentObj of INTENTS) {
  for (const kw of intentObj.keywords) {
    if (!KEYWORD_REGEXES.has(kw)) {
      KEYWORD_REGEXES.set(kw, new RegExp(`\\b${kw.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')}\\b`, 'i'));
    }
  }
}
 
// Helper functions to match your Python version
function keywordFound(text: string, kw: string): boolean {
  // Use cached regex for fast lookup
  const regex = KEYWORD_REGEXES.get(kw);
  if (regex) {
    return regex.test(text);
  }
  // Fallback (shouldn't happen with pre-compiled cache)
  return text.includes(kw.toLowerCase());
}

    params['measurement'] = {
      value: parseFloat(unitMatch[1].replace(/,/g, '')),
      unit: unitMatch[2].toLowerCase()
    };
  }
 
  return params;
}
 
// Helper to check if a phrase (multi-word keyword) appears exactly in text
function exactPhraseMatch(text: string, phrase: string): boolean {
  const regex = new RegExp(`\\b${phrase.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')}\\b`, 'i');
  return regex.test(text);
}
 
// Helper to calculate keyword specificity score
function getKeywordWeight(keyword: string, text: string): number {
  const wordCount = keyword.split(/\s+/).length;
  
  // Base weight: multi-word phrases are more specific
  let weight = wordCount * 3;
  
  // Bonus for exact phrase match
  if (wordCount > 1 && exactPhraseMatch(text, keyword)) {
    weight += 5; // Strong signal for exact multi-word match
  }
  
  // Penalize generic single-word keywords
  const genericKeywords = ['show', 'list', 'display', 'find', 'get', 'fetch', 'view', 'see', 'data'];
  if (wordCount === 1 && genericKeywords.includes(keyword)) {
    weight = 1; // Much lower weight for generic keywords
  }
  
  // Boost discriminator keywords that help distinguish specific intents
  const discriminators = ['total tonnage', 'total trips', 'average production', 'monthly report', 
                          'shift rank', 'equipment breakdown', 'production summary'];
  if (discriminators.includes(keyword)) {
    weight += 4;
  }
  
  return weight;
}
 
export function normalizeUserQuery(userInput: string): IntentResult {
  const text = userInput.toLowerCase();
  
  // Track all matching intents with their scores
  const candidateIntents: Array<{
    intent: string;
    score: number;
    matchedKeywords: string[];
    tier: IntentTier;
  }> = [];
 
  // Score all intents
  for (const intentObj of INTENTS) {
    let score = 0;
    const matches: string[] = [];
    
    for (const kw of intentObj.keywords) {
      if (keywordFound(text, kw)) {
        const weight = getKeywordWeight(kw, text);
        score += weight;
        matches.push(kw);
      }
    }
    
    // Only consider intents that matched at least one keyword
    if (score > 0) {
      candidateIntents.push({
        intent: intentObj.intent,
        score,
        matchedKeywords: matches,
        tier: intentObj.tier
      });
    }
  }
 
  // If no matches, return UNKNOWN
  if (candidateIntents.length === 0) {
    return {
      intent: 'UNKNOWN',
      confidence: 0,
      matched_keywords: [],
      parameters: extractParameters(userInput),
    };
  }
 
  // Apply tier-based filtering: remove generic intents if specific ones matched
  const hasSpecificMatches = candidateIntents.some(c => c.tier === IntentTier.TIER_1_SPECIFIC);
  const hasModerateMatches = candidateIntents.some(c => c.tier === IntentTier.TIER_2_MODERATE);
  
  let filteredCandidates = candidateIntents;
  
  // If we have TIER_1 matches, exclude TIER_3
  if (hasSpecificMatches) {
    filteredCandidates = candidateIntents.filter(c => c.tier !== IntentTier.TIER_3_GENERIC);
  }
  // If we have TIER_2 matches but no TIER_1, still exclude TIER_3
  else if (hasModerateMatches) {
    filteredCandidates = candidateIntents.filter(c => c.tier !== IntentTier.TIER_3_GENERIC);
  }
 
  // Mutual exclusion rules: specific aggregation intents override AGGREGATION_QUERY
  const specificAggregationIntents = ['GET_TOTAL_TONNAGE', 'GET_TOTAL_TRIPS', 'GET_AVERAGE_PRODUCTION', 'GET_MONTHLY_SUMMARY'];
  const hasSpecificAggregation = filteredCandidates.some(c => specificAggregationIntents.includes(c.intent));
  if (hasSpecificAggregation) {
    filteredCandidates = filteredCandidates.filter(c => c.intent !== 'AGGREGATION_QUERY');
  }
 
  // Sort by multiple criteria (tiebreaker logic)
  filteredCandidates.sort((a, b) => {
    // 1. Higher score wins
    if (b.score !== a.score) return b.score - a.score;
    
    // 2. Specific tier wins (lower tier number is higher priority)
    if (a.tier !== b.tier) return a.tier - b.tier;
    
    // 3. More keywords matched wins
    if (b.matchedKeywords.length !== a.matchedKeywords.length) {
      return b.matchedKeywords.length - a.matchedKeywords.length;
    }
    
    // 4. Longer total keyword length wins (more specific)
    const aLen = a.matchedKeywords.join('').length;
    const bLen = b.matchedKeywords.join('').length;
    if (bLen !== aLen) return bLen - aLen;
    
    // 5. Alphabetical order as final fallback (deterministic)
    return a.intent.localeCompare(b.intent);
  });
 
  const bestMatch = filteredCandidates[0];
  
  // Calculate confidence based on score and tier
  // Tier-specific confidence scaling
  let maxScore = 20; // Expected max score for high-quality match
  if (bestMatch.tier === IntentTier.TIER_1_SPECIFIC) {
    maxScore = 18; // Specific intents tend to have fewer but more precise keywords
  } else if (bestMatch.tier === IntentTier.TIER_2_MODERATE) {
    maxScore = 20; // Moderate intents have good keyword coverage
  } else {
    maxScore = 25; // Generic intents match broadly
  }
  
  const confidence = Math.min(1.0, bestMatch.score / maxScore);
  const params = extractParameters(userInput);
 
  return {
    intent: bestMatch.intent,
    confidence: Math.round(confidence * 100) / 100,
    matched_keywords: bestMatch.matchedKeywords,
    parameters: params,
  };
}
 
export default normalizeUserQuery;
